    "Load all Emacs Lisp files in the specified directory DIR."
    (when (file-directory-p dir)  ;; Check if dir is a valid directory
      (dolist (file (directory-files dir t "\\.el$"))  ;; `t` makes the file names absolute
        (when (file-regular-p file)  ;; Ensure it's a regular file

    "Load the Emacs Lisp file FILENAME from directory DIR if it exists and is a valid .el file."
    (let ((filepath (expand-file-name filename dir)))
      (if (and (file-regular-p filepath) (string= (file-name-extension filepath) "el"))
          (load-file filepath)
        (message "File %s does not exist or is not a valid .el file" filepath))))

*** macro
#+begin_src emacs-lisp
  (defvar temc/macros-file (expand-file-name "macros.el" user-emacs-directory)
    "File where named macros are stored.")

  (defun temc/macro-save-named (macro-name)
    "Save a named macro to the macro file.
  Interactively, prompt for the macro name."
    (interactive "sMacro name: ")
    (let* ((macro-symbol (intern macro-name))
           (macro-def (and (fboundp macro-symbol)
                           (get macro-symbol 'kmacro)
                           (symbol-function macro-symbol))))
      (if macro-def
          (progn
            ;; Create directory if it doesn't exist
            (unless (file-exists-p (file-name-directory temc/macros-file))
              (make-directory (file-name-directory temc/macros-file) t))

            ;; Save the macro
            (with-temp-buffer
              (emacs-lisp-mode)
              (insert ";; -*- emacs-lisp -*-\n")
              (insert ";; Saved macros - DO NOT EDIT MANUALLY\n\n")

              ;; Load existing macros if the file exists
              (when (file-exists-p temc/macros-file)
                (insert-file-contents temc/macros-file))

              ;; Remove existing definition if present
              (goto-char (point-min))
              (let ((case-fold-search nil))
                (while (re-search-forward
                        (format "^(fset '%s\n" (regexp-quote macro-name))
                        nil t)
                  (forward-line -1)
                  (let ((start (point)))
                    (forward-sexp 1)
                    (delete-region start (point)))))

              ;; Add the new macro at the end
              (goto-char (point-max))
              (insert (format "\n;; Macro: %s\n" macro-name))
              (insert (format "(fset '%s\n" macro-name))
              (insert (format "   %S)\n" macro-def))

              ;; Write the file
              (write-region (point-min) (point-max) temc/macros-file))

            (message "Saved macro '%s' to %s" macro-name temc/macros-file))
        (message "No macro named '%s' found. Create one with 'M-x kmacro-name-last-macro' first." macro-name))))

  (defun temc/macros-load ()
    "Load saved macros from the macro file."
    (when (file-exists-p temc/macros-file)
      (load-file temc/macros-file)
      (message "Loaded macros from %s" temc/macros-file)))
#+end_src

*** stop
Function method to stop properly emacs background instance.
#+begin_src emacs-lisp
  (defun stop-save-kill-emacs ()
    " Stop the emacs daemon.
      It can be used to save buffers and shutdown emacs.
      It should be called using emacsclient -e '(stop-save-kill-emacs)'.
      This function will check to see if there are any modified buffers
      or active clients or frame.
      If so an x window will be opened and the user will be prompted."
    (interactive)
    (let (new-frame modified-buffers active-clients-or-frames)

      ;; Check if there are modified buffers or active clients or frames.
      (setq modified-buffers (modified-buffers-exist))
      (setq active-clients-or-frames ( or (> (length server-clients) 1)
                                       (> (length (frame-list)) 1)
                                       ))

      ;; When displaying the number of clients and frames:
      ;; subtract 1 from the clients for this client.
      ;; subtract 2 from the frames this frame (that we just created) and the default frame.
      (when ( or (not active-clients-or-frames)
              (yes-or-no-p (format "There are currently %d clients and %d frames. Exit anyway?" (- (length server-clients) 1) (- (length (frame-list)) 2))))

        ;; If the user quits during the save dialog then don't exit emacs.
        ;; Still close the terminal though.
        (let((inhibit-quit t))
          ;; Save buffers
          (with-local-quit
            (save-some-buffers))

          (if quit-flag
              (setq quit-flag nil)
            ;; Kill all remaining clients
            (progn
              (dolist (client server-clients)
                (server-delete-client client))
              ;; Exit emacs
              (kill-emacs)))
          ))
      )
    )

  (defun modified-buffers-exist()
    "This function will check to see if there are any buffers
      that have been modified.  It will return true if there are
      and nil otherwise. Buffers that have buffer-offer-save set to
      nil are ignored."
    (let (modified-found)
      (dolist (buffer (buffer-list))
        (when (and (buffer-live-p buffer)
                   (buffer-modified-p buffer)
                   (not (buffer-base-buffer buffer))
                   (or
                    (buffer-file-name buffer)
                    (progn
                      (set-buffer buffer)
                      (and buffer-offer-save (> (buffer-size) 0))))
                   )
          (setq modified-found t)
          )
        )
      modified-found
      )
    )
#+end_src

      "Run npm script for current TypeScript project."

      (let* ((package-json (locate-dominating-file default-directory "package.json"))
             (project-dir (if package-json (file-name-directory package-json) default-directory)))

          (compile (format "cd %s && npm run %s" (shell-quote-argument project-dir) script)))))
